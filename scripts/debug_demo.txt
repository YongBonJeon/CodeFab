//   1. step  으로 한 줄씩 실행
//   2. watch n  으로 변수 n 감시
//   3. inspect  으로 전체 변수 확인
//   4. break 14  으로 breakpoint 설정 후 continue

Func fact(n) {
    if (n <= 1) return 1;
    return n * fact(n - 1);
}

var result = fact(5);
print result;

var arr = Array(3);
arr[0] = 10;
arr[1] = 20;
arr[2] = 30;

var sum = 0;
for (var i = 0; i < 3; i = i + 1) {
    sum = sum + arr[i];
}
print sum;
